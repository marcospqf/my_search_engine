			var page = "";
			if ( typeof window.paywall_after === "function" ) {
				window.paywall_after( data ) ;
			}

			if ( typeof data.status == "undefined" || data.status != "ok" || data.paywall != "on" ) {
				return false ;
			}

			if( !pattern_path.test( window.location ) ) {
				if ( data.log_type !== "subscriber" && ( typeof folha_ads === undefined || folha_ads !== true ) ) {
					window.location.replace( 'http://www1.folha.uol.com.br/paywall/adblock.shtml?origin=after&url=' + paywall.url() ) ;
				}
			}

				return false ;

				if ( data.skin != "default" ) {
