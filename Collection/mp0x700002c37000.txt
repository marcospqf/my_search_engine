function getPostIdClass(elem) {
	var classes = jQuery(elem).attr('class').split(/\s+/),
	    id = 0;

	for ( var i = 0; i < classes.length; i++ ) {
		if ( classes[i].match(/^post\-/i) ) {
			id = classes[i];
			break;
		}
	}

	return id;
