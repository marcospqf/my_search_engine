Ext.onReady(function () {

  var win = new Ext.Window({
    layout: 'fit',
    title: 'Reset Password',
    width: 500,
    height: 300,
    modal: true,
    closeAction: 'hide',

    resetForm: function() {
      this.mailField.clearInvalid();
    },

    items: [
      {
        xtype: 'container',
        layout: 'form',
        items: [
          {
            // just a placeholder
            xtype: 'container',
            height: 20
          },
          {
            xtype: 'textfield',
            allowBlank: false,
            width: 300,
            ref: '../mailField',
            fieldLabel: 'E-Mail'
          }
        ]
      }
    ],

    buttons: [
      {
        text: 'Back',
        listeners: {
          scope: this,
          click: function() {
            win.hide();
          }
        }
      },
      {
        text: 'Send',
        ref: 'send_btn',
        listeners: {
          click: function() {
            if (!win.mailField.validate()) {
              Toast.showMessage('Please enter your email address.', false);
            } else {
              this.disable();

              var that = this;

              Ext.Ajax.request
              ({
                url: '/resetPassword',
                method: 'POST',
                params: {
                  email: win.mailField.getValue()
                },
                success: function(response)
                {
                  that.enable();
                  var response = Ext.util.JSON.decode(response.responseText);

                  Toast.showMessage(response.msg);
                  if (response.success == 1) {
                    win.hide();
                  }
                },
                failure: function(response)
                {
                  that.enable();
                  Toast.showMessage("Failed to reset password.");
                }
              });
            }
          }
        }
      }
    ]
  });

  var link = Ext.get('forgot-pw');
  link.addListener('click', function () {
    var mail = Ext.get('email').getValue();
    win.mailField.setValue(mail);
    win.on('show', win.resetForm, win);
    win.show(this);
  }, this);

});
