function getCookie(cname) {
    var name = cname + "=";
    var ca = document.cookie.split(';');
    for(var i = 0; i <ca.length; i++) {
        var c = ca[i];
        while (c.charAt(0)==' ') {
            c = c.substring(1);
        }
        if (c.indexOf(name) == 0) {
            return c.substring(name.length,c.length);
        }
    }
    return false;
}

function parse_date(str) {
    var es_date_format = /\d\d\/\d\d\/\d\d\d\d/;
    var splitdate;

    if (str.match(es_date_format)) {
        str = str.replace(/\//g, '-');

function goto_widget_results() {

    window.location = url;

function get_slug( idcar ) {
    return $('#' + idcar).data('slug');
}

function build_search_url() {
    var url = searchPath;
    // copy all form params
    // instead of checked carNNN fields, use the related slug inside a new 'chars' parameter.

    var chars = "";
    var other = "";

    var params = $('#filtersForm').serializeArray();
    var chars_re = /^(car\d+)/;
    var skip_re = /^(idgeo|idzona)$/;

    $.each( params,
        function(index, param) {
            var matches;
            if ( matches = chars_re.exec( param.name ) ) {
                var idcar = matches[0];
                if (chars.length > 0) { chars += '+'; }
                var slug = get_slug( idcar );
                chars += slug;
            }
            // other wanted parameters
            else if (! skip_re.exec( param.name )) {
                if (param.value != '') {
                    if (other.length > 0) { other += '&'; }
                    other += param.name + '=' + param.value;
                }
            }
        }
    );

    url += other;
    if (chars.length > 0) {
        url += '&chars=' + chars;
    }

    // add geo/zone
    if ( $('#idzona_copy').val().length != 0 ) {
        // special geo for zones
        url += "&idgeo=" + "9";
    }
    else {
        url += "&idgeo=" + $('#idgeo_copy').val();
    }
    url += "&idzona=" + $('#idzona_copy').val();

    return url;
}

// replaces a select element with a custom dropdown with manual-entry text field
function setup_dropdown( element, options ) {

    var dropdown_items = $.map(
        element.find('option'),
        function(item) {
            var obj = $(item);
            return { val: obj.val(), text: obj.text(), selected: obj.attr('selected') };
        }
    );
    var input_id = element.attr('id') + '_value';

    var handler_options = {
        element_id: element.attr('id'),
        input_id: input_id,
        items: dropdown_items,
        change: options.change ? options.change : null,
        render: options.render ? options.render : null,
        curval: options.curval

    };


    var tmpl = $('#dropdown_template').tmpl( handler_options );

    element.replaceWith( tmpl );

    install_dropdown_handler( handler_options );

}

function preventSubmit(ev) {
    if (ev.keyCode == 13) {
        ev.preventDefault();
        ev.target.blur();
        return false;
    }
};

function update_price_selectors() {
    val = $('#tpre').val();

        $('#daily_price_select').hide();

            $('#daily_price_select').show();

        else {
            $('#daily_price_select').hide();
            $('#weekly_price_select').show();
            $('#max_price').val( $('#weekly_price_select_value').val() );
        }
    }
}

function install_dropdown_handler( options ) {
    var element = $('#' + options.element_id);
    var hidden_input = $('#' + options.input_id);
    var value_display = $('#' + options.element_id + '_display');
    var dropdown_list = $( element.find('.dropdown_list') );
    var manual_input = $('#' + options.element_id + '_SELECT_INPUT');
    var render = options.render != null
        ? options.render
        : function (val) {
            return "[" + val + "]";
        };

    // keep label synced with hidden val
    // hide list on element select or manual input change
    hidden_input.change( function(e) {
        value_display.text( render (hidden_input.val(), hidden_input.attr('rel')) );
        if (options.change) {
            options.change(e);
        }
    });

    // keep hidden val in sync with input box
    manual_input.change( function(e) {
        hidden_input.attr('rel', manual_input.attr('rel') );
        hidden_input.val( manual_input.val() );
        dropdown_list.hide();
        hidden_input.change();
        return false;
    });

    // manually typed values have no other textual display, so empty it
    manual_input.keypress( function(e) {
        manual_input.attr('rel', '');
    });

    // click on value -> update input box
    $( dropdown_list.find('a.dropdown_item') ).click(
        function(ev) {
            var target = $(ev.target);
            manual_input.val( target.attr('rel') );
            manual_input.attr('rel', target.text() );
            manual_input.change();
        }
    );

    // click to show/hide value list
    $( element.find('a.dropdown_expander') ).click(function () {
        dropdown_list.toggle();
    });

    // prevent form submission by pressing enter
    manual_input.keypress(preventSubmit);
    manual_input.keydown(preventSubmit);
    manual_input.keyup(preventSubmit);

    // at start, select current value if existing
    manual_input.attr('rel', '');
    manual_input.val( options.curval );
    manual_input.change();
}
