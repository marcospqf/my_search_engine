            .replace("ú", "u")

                current++;
                var cls = replaceSpecialCharacters($li.eq(current).find('a').text().toLowerCase());
                $ul.removeAttr('class').addClass(cls);
            }
        });

        $('.readMore > .navigation > .back').click(function() {
