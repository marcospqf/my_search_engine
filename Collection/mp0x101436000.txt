	function formatarTempo(d, id){
	seconds = Math.floor(d / 1000);
	minutes = Math.floor(seconds / 60);
	timeTemp = minutes * 60;
	seconds = seconds - timeTemp;
	if(minutes < 10){
	minutes = "0"+minutes;
	}
	if(seconds < 10){
	seconds = "0"+seconds;
	}
	tempoFinal = minutes +':'+ seconds;
