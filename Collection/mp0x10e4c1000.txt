				switch (etapa) {
					case 0: /* Nada faz. Etapa transitória, temporária */; break;
					case 1: etapa1(); break;
					case 2: etapa2(); break;
					default: alert("Houve um ERRO ao acessar o sistema"); break;
				}
			}
			
			// Qualquer tecla
			if (etapa == 2) {
				$("input[name='dosInput']").focus();
			}
			
		}

		$(".dosInputShow").html($(this).val());

	function etapa1() {
		$(".cursorInicial").hide();
		
		setTimeout(function () {
			$("#dosOutput").append(texto[i]);
			i++;
			if (i < texto.length) {
				etapa1();
			}
		}, 250)
		
		$("input[name='dosInput']").focus();
		
		etapa = 0;
		
		// Ao terminar de mostrar todo texto no DOS
		if (i == (texto.length - 1)) {
			etapa = 2;
			$("input[name='dosInput']").val("");
		}
	}
	
	
	/** ============ ETAPA 2 ============ **/	
	function etapa2() {
		userInput = $("input[name='dosInput']").val();
		userInput = userInput.toLowerCase();
		userInput = userInput.replace(/\s/g, ''); // Remove espaços em branco

			$("#dosOutput").append("Carregando......<br />");
			window.location.href = '/pt/trabalhos';
		}
		
		else if (userInput == "3") {
			$("#dosOutput").append("Carregando......<br />");
			window.location.href = '/pt/servicos';
		}
		
		else if (userInput == "4") {
			$("#dosOutput").append("Carregando......<br />");
			window.location.href = '/pt/clientes';
		}
		
		else if (userInput == "4") {
			$("#dosOutput").append("Carregando......<br />");
			window.location.href = '/pt/produtos';
		}
		
		else if (userInput == "5") {
			$("#dosOutput").append("Carregando......<br />");
			window.location.href = 'http://blog.riandesign.com.br';
		}
		
		else if (userInput == "6") {
			$("#dosOutput").append("Carregando......<br />");
			window.location.href = '/pt/contato';

			$.magnificPopup.close();

			resolveEquacao("-");
		}
		else if (userInput.indexOf("*") > -1) {

	/** ============ FUNCOES DOS ============ **/
	function inicializaDos() {
		dosAberto = true;
		userInput = '';
		etapa = 1;
		i = 0;
		$(".cursorInicial").show();
		$("#dosOutput").html("");
	}
	
	function resolveEquacao(sinal) {
		equacao = userInput.split(sinal);
		n1 = parseInt(equacao[0]);
		n2 = parseInt(equacao[1]);
		total = 0;
		switch (sinal) {
			case "+": total = n1 + n2; break;
			case "-": total = n1 - n2; break;
			case "*": total = n1 * n2; break;
			case "/": total = n1 / n2; break;
		}
		$("#dosOutput").append(total + "<br />");
	}
	
	// Pisca "cursor"
	(function blink() {
		$(".blinking").fadeOut(250).fadeIn(250, blink); 
	})();
	
});
