function addEventVideo() {
    var video = document.getElementsByTagName('video')[0];
    video.addEventListener("play", function(e) {
        _gaq.push(["_trackEvent","Video iniciado: " + ((video.getAttribute('title') != null)?video.getAttribute('title'):"Previsao Diaria"), e.type, video.currentSrc]);
    }, false);
    video.addEventListener("ended", function(e) {
        _gaq.push(["_trackEvent","Video finalizado: " + ((video.getAttribute('title') !== null)?video.getAttribute('title'):"Previsao Diaria"), e.type, video.currentSrc]);
    }, false);
}
