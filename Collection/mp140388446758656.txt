                function reloadCaptcha(captcha) {
                    $('[name="captcha_0"]').val(captcha['new_key']);
                    $('[name="captcha_1"]').val('');
                    $('.captcha').attr('src', captcha['new_image']);
                }

                var feedbackForm = $('#feedback-form');
                $('.showFeedBackPopup').on('click', function(){
                    $('.blind').show();

                    feedbackForm.show();
                    $('#feedback .prog-ok-message').hide();

                    $.get("/ajax/get_captcha/", function( response ) {
                        var data = $.parseJSON(response);
                        reloadCaptcha(data);
                    });
                    $('.close').click(function(){
                        $('.blind').hide();
                    });
                });

                feedbackForm.on('submit', function(e){
                    e.preventDefault();
                    var data = {};
                    feedbackForm.find('.formError').html('');

                    $.post(serverUrls['ajax_feedback_form'], {
                        csrfmiddlewaretoken: $.cookie('csrftoken'),
                        name: $('[name="name"]').val(),
                        email: $('[name="email"]').val(),
                        subject: $('[name="subject"]').val(),
                        message: $('[name="message"]').val(),
                        captcha_0: $('[name="captcha_0"]').val(),
                        captcha_1: $('[name="captcha_1"]').val()
                    }, function(response) {

                        data = $.parseJSON(response);
                        reloadCaptcha(data);

                        if (data.status == '200') {
                            $('#feedback .prog-ok-message').show();
                            feedbackForm.hide();
                            feedbackForm.find('input, textarea').each(function () {
                                $(this).val('');
                            });
                        }
                        else if (data.status == '403') {
                            feedbackForm.find('.formError').html(data.errors[0]);
                        }
                        else if (data.status == '500') {
                            $('#feedback-form .error').removeClass('error');
                            for(var key in data.errors) {
                                $('#feedback-form [name=' + data.errors[key] + ']').addClass('error');
                                if (data.errors[key] == 'captcha')
                                    $("#feedback-form input[name=captcha_1]").addClass('error');
                            }
                        }
                    });
                });
