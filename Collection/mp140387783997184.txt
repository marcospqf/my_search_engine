        function replaceSpecialCharacters(str) {
            return str
            .replace("á", "a")
            .replace("é", "e")
            .replace("í", "i")
            .replace("ó", "o")
            .replace("ú", "u")
            .replace("ç", "c")
            .replace("à", "a")
            .replace("ã", "a")
            .replace("õ", "o")
        }

        $('.readMore > .navigation > .next').click(function() {
            if (current < $li.size()-1) {
                current++;
                var cls = replaceSpecialCharacters($li.eq(current).find('a').text().toLowerCase());
                $ul.removeAttr('class').addClass(cls);
            }
        });

        $('.readMore > .navigation > .back').click(function() {
            if (current > 0) {
                current--;
                var cls = replaceSpecialCharacters($li.eq(current).find('a').text().toLowerCase());
                $ul.removeAttr('class').addClass(cls);
            }
        });


        $('.carouselLidasContainer li').eq(start).each(function () {
            var element = $(this).find('a').text().toLowerCase();
            var cls = replaceSpecialCharacters(element);
            $(this).parent().removeAttr('class').addClass('carouselLidasContainer ' + cls);
        });



        /* Paginadores */
        $('.carouselEditorias').paginar({
            start : current
        });
