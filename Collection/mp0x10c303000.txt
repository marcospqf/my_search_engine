  function fixMenuOnScroll() {
      var menu = $('#menu');
      var geral = $('#geral');
      var offSetMenu = 118;
      $(window).scroll(function () {
          var offSetWindow = $(window).scrollTop();
          if (offSetWindow >= offSetMenu) {
              menu.addClass('menu-fixed');
              geral.addClass('geral-absolute');
          } else {
              menu.removeClass('menu-fixed');
              geral.removeClass('geral-absolute');
          }
      });
  }
  
  fixMenuOnScroll();
  
  var calculatadeMenu = false;

  function larguraSubmenus() {
      var aba, paddingR, totalAba, container, retorno;
      $('#menu li.sub').each(function () {
          aba = $(this).find('.aba').width();
          paddingR = $(this).find('.aba').css('padding-right').replace('px', '');
          totalAba = aba + parseInt(paddingR);
          container = $(this).find('.conteudoSubMenu').width();
          retorno = container - totalAba;
          if ($(this).hasClass('last')) {
              if (($(this).position().left + $(this).find('.conteudoSubMenu').width()) > 988) {
                  $(this).find('.conteudoSubMenu').css('left', function () {
                      return '-' + retorno + 'px'
                  });
              }
          }
      });
      calculatadeMenu = true;
  }
  $('#buscar').focus(function () {
      $('#buscaGoogle').addClass('getFocus');
  });
  $('#buscar').blur(function () {
      $('#buscaGoogle').removeClass('getFocus');
  });
  $('#menu h2 a').hover(function () {
      if (!calculatadeMenu) {
          larguraSubmenus();
      }
      $(this).closest('li').find('.menuContainer').css({
          visibility: "visible"
      }).animate({
          opacity: 1
      }, 50, function () {});
  }, function () {
      openedMenu = setTimeout(function (obj) {
          $('.menuContainer').animate({
              opacity: 0
          }, 50, function () {
              $(this).css({
                  visibility: "hidden"
              })
          })
      }, 300);
  });
  $('#menu .menuContainer').hover(function () {
      $(this).closest('li').find('.menuContainer').css({
          visibility: "visible",
          opacity: 1
      });
      clearInterval(openedMenu);
  }, function () {
      $(this).animate({
          opacity: 0
      }, 50, function () {
          $(this).css({
              visibility: "hidden"
          })
      })
  });

  function fixMenuOnScroll() {
      var menu = $('#menu');
      var geral = $('#geral');
      var offSetMenu = 118;
      $(window).scroll(function () {
          var offSetWindow = $(window).scrollTop();
          if (offSetWindow >= offSetMenu) {
              menu.addClass('menu-fixed');
              geral.addClass('geral-absolute');
          } else {
              menu.removeClass('menu-fixed');
              geral.removeClass('geral-absolute');
          }
      });
  }
  
  fixMenuOnScroll();
  
  var calculatadeMenu = false;

  function larguraSubmenus() {
      var aba, paddingR, totalAba, container, retorno;
      $('#menu li.sub').each(function () {
          aba = $(this).find('.aba').width();
          paddingR = $(this).find('.aba').css('padding-right').replace('px', '');
          totalAba = aba + parseInt(paddingR);
          container = $(this).find('.conteudoSubMenu').width();
          retorno = container - totalAba;
          if ($(this).hasClass('last')) {
              if (($(this).position().left + $(this).find('.conteudoSubMenu').width()) > 988) {
                  $(this).find('.conteudoSubMenu').css('left', function () {
                      return '-' + retorno + 'px'
                  });
              }
          }
      });
      calculatadeMenu = true;
  }
  $('#buscar').focus(function () {
      $('#buscaGoogle').addClass('getFocus');
  });
  $('#buscar').blur(function () {
      $('#buscaGoogle').removeClass('getFocus');
  });
  $('#menu h2 a').hover(function () {
      if (!calculatadeMenu) {
          larguraSubmenus();
      }
      $(this).closest('li').find('.menuContainer').css({
          visibility: "visible"
      }).animate({
          opacity: 1
      }, 50, function () {});
  }, function () {
      openedMenu = setTimeout(function (obj) {
          $('.menuContainer').animate({
              opacity: 0
          }, 50, function () {
              $(this).css({
                  visibility: "hidden"
              })
          })
      }, 300);
  });
  $('#menu .menuContainer').hover(function () {
      $(this).closest('li').find('.menuContainer').css({
          visibility: "visible",
          opacity: 1
      });
      clearInterval(openedMenu);
  }, function () {
      $(this).animate({
          opacity: 0
      }, 50, function () {
          $(this).css({
