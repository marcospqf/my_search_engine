    var renderMenuHighlights = function (strEditoria, element, destacado) {

    }

    var renderMenuSenarHighlights = function (strEditoria) {

        var photoTemplate = '<aside class="imagem">' +

            if (editoria) {
                $(editoria.conteudos).each(function (index, item) {

    function string_to_slug(str) {
        str = str.replace(/^\s+|\s+$/g, ''); // trim

        // remove accents, swap ñ for n, etc
        var from = "ÃÁÉÍÓÚÇãàáäâèéëêìíïîòóöôõùúüûñç·/_,:;";
        var to   = "AAEIOUCaaaaaeeeeeiiiiooooouuuunc------";

        for (var i=0, l=from.length ; i<l ; i++) {
            str = str.replace(new RegExp(from.charAt(i), 'g'), to.charAt(i));
        }

        str = str.replace(/[^A-Za-z0-9 -]/g, '') // remove invalid chars
        str = str.replace(/\s+/g, '-') // collapse whitespace and replace by -
        str = str.replace(/-+/g, '-'); // collapse dashes

        return str.toLowerCase();
    }

    function truncar(texto,limite){
        if(texto.length>limite){
            limite--;
            last = texto.substr(limite-1,1);
            while(last!=' ' && limite > 0){
                limite--;
                last = texto.substr(limite-1,1);
            }
            last = texto.substr(limite-2,1);
            if(last == ',' || last == ';'  || last == ':'){
                 texto = texto.substr(0,limite-2) + '...';
            } else if(last == '.' || last == '?' || last == '!'){
                 texto = texto.substr(0,limite-1);
            } else {
                 texto = texto.substr(0,limite-1) + '...';
            }
        }
        return texto;
    }

    var informacoes_do_produto_pequeno = ""

    $(document).ready(function () {

        $.getJSON("cotacoes/agronegocio/cambio/select/", function(data){
            jQuery(".dolar-valor span").html(data[0].compra)
        })

        var dataProduto = []
        var json = "cotacoes/agronegocio_produtos/futuro/";
        var indexSoja = 0;
        
        $.getJSON(json, function(data){    
            $.each(data, function(i){
                dataProduto.push({id:data[i].id, value:data[i].nome, label:data[i].nome});
                if (data[i].nome == "Soja")
                    indexSoja = i;
             });
        }).done(function(){
            produto_randomico = dataProduto[indexSoja];
            $( "#select-produto" ).attr("value",produto_randomico.label);
            popula_informacoes_do_produto_pequeno(produto_randomico.id);
        });    

        $( "#select-produto" ).autocomplete({
        source: dataProduto,
        minLength: 0,
        valor_selecionado_anteriormente: this.value,
        messages: {
            noResults: '',
            results: function() {}
        },
        select: function (event, ui) {
            popula_informacoes_do_produto_pequeno(ui.item.id);
            $('#select-produto').blur();
        }
        }).focus(function(){
            valor_selecionado_anteriormente = this.value
            this.value = ""
            if (this.value == "")
            {
                $(this).autocomplete("search");
            }
        }).focusout(function(){
            this.value = valor_selecionado_anteriormente;
        });

        function popula_informacoes_do_produto_pequeno(id_produto){
            var json = "cotacoes/agronegocio_bolsa_por_produto/futuro/"+id_produto+"/";
            $.getJSON(json, function(data){
                informacoes_do_produto_pequeno = data;
            }).done(function() {
                monta_select_bolsa_pequeno();
                if ( informacoes_do_produto_pequeno.bolsas.length > 0 ) {
                    monta_vencimentos_mercado_futuro_pequeno(informacoes_do_produto_pequeno.bolsas[0].descricao);
                    monta_select_papel_pequeno(informacoes_do_produto_pequeno.bolsas[0].descricao);
                    if ( informacoes_do_produto_pequeno.bolsas[0].papeis.length > 0 ){
                        drawVisualizationPequeno(informacoes_do_produto_pequeno.bolsas[0].papeis[0].papel);
                    }
                }
            });
        }

        function monta_select_bolsa_pequeno (){
            var dataBolsa = []
            bolsas = informacoes_do_produto_pequeno.bolsas;
            $.each(bolsas, function(i){
                dataBolsa.push({id:1, value:bolsas[i].descricao, label:bolsas[i].descricao});
            });
            
            $( "#select-bolsa" ).attr("value",dataBolsa[0].label);

            var valor_selecionado_anteriormente = dataBolsa[0].label;
            
            $( "#select-bolsa" ).autocomplete({
            source: dataBolsa,
            minLength: 0,
            messages: {
                noResults: '',
                results: function() {}
            },
            select: function (event, ui) {
                monta_vencimentos_mercado_futuro_pequeno(ui.item.value);  
                monta_select_papel_pequeno(ui.item.value);
                valor_selecionado_anteriormente = ui.item.value;
                $('#select-bolsa').blur();
            }
            }).focus(function(){
                this.value = ""
                if (this.value == "")
                {
                    $(this).autocomplete("search");
                }
            }).focusout(function(){
                this.value = valor_selecionado_anteriormente;
            });
        }

        function monta_select_papel_pequeno (bolsa){
            bolsas = informacoes_do_produto_pequeno.bolsas;
            var items = [];

            $.each(bolsas, function(i){
                if (bolsas[i].descricao == bolsa)
                {       
                    papeis = bolsas[i].papeis;            
                    $.each(papeis, function(i){    
                        papel_data_contrato = papeis[i].descricao.split("-")[0].substring(1, 4) + " " + papeis[i].descricao.split("-")[1];
                        item = '<div><span data="'+ papeis[i].papel +'"class="">'+  papel_data_contrato  + '</span></div>';
                        items.push( item );
                    })
                    
                    drawVisualizationPequeno(papeis[0].papel);

                    $('.contrato').html(items);
                }
            });

            jQuery(jQuery('.contrato span')[0]).addClass( "selecionado" );
            jQuery('.contrato span').bind( "click", function() {
              removeTodosSelecionados();
              $( this ).addClass( "selecionado" );
              drawVisualizationPequeno($( this ).attr( "data" ));
            });
        }

        function removeTodosSelecionados(){
          jQuery('.contrato span').each( function() {
            $( this ).removeClass( "selecionado" );
          });
        }

        function monta_vencimentos_mercado_futuro_pequeno (bolsa){
            //var dataPapel = []
            bolsas = informacoes_do_produto_pequeno.bolsas;
            var items = [];
            $.each(bolsas, function(i){
                if (bolsas[i].descricao == bolsa)
                {
                    papeis = bolsas[i].papeis
                    $.each(papeis, function(i){
                        var tipo_numero = "positvo";

                        if (papeis[i].variacao < 0) {
                            tipo_numero = "negativo";
                        }
            
                        //dataPapel.push({id:1, value:papeis[i].papel, label:papeis[i].papel});

                        mes = papeis[i].descricao.split("-")[0]
                        ano = "20" + papeis[i].descricao.split("-")[1]

                        tr ='<tr class="linha">' +
                            '<td class="coluna"><span>'+mes+' '+ano+'</span></td> ' +
                            '<td class="coluna ultimo"><span>'+ papeis[i].ultimo +'</span></td> ' +
                        '</tr>';   
                        items.push( tr );                 
                    })
                    
                    // jQuery(".identificador-moeda-futuro").each(function(){

                    // })

                }    
            });
            $('.t-abertos').html(items);
        }

        function drawVisualizationPequeno(papel) {
            historicos = [];
            bolsas = informacoes_do_produto_pequeno.bolsas;
            $.each(bolsas, function(i){
                papeis = bolsas[i].papeis
                $.each(papeis, function(i){
                    if (papeis[i].papel == papel)
                    {
                        historicos = papeis[i].historicos;
                    }
                });
            });

            
            data_final_cortada = historicos[historicos.length - 1].data.split("/");

            var indice_data_corte = 4

            if ( historicos.length  < indice_data_corte)
                data_inicial_cortada = historicos[historicos.length - 1].data.split("/");
            else
                data_inicial_cortada = historicos[historicos.length - indice_data_corte].data.split("/");
            

            var dashboard = new google.visualization.Dashboard(document.getElementById('grafico-pequeno'));

            var control = new google.visualization.ControlWrapper({
             'controlType': 'ChartRangeFilter',
             'containerId': 'control-pequeno',
             'options': {
               // Filter by the date axis.
               'filterColumnIndex': 0,
               'ui': {
                 'chartType': 'LineChart',
                 'chartOptions': {
                   'chartArea': {'height': '50px', 'width': '273px'},
                   'hAxis': {'baselineColor': 'none'}
                 },
                 // Display a single series that shows the closing value of the stock.
                 // Thus, this view has two columns: the date (axis) and the stock value (line series).
                 'chartView': {
                   'columns': [0, 1]
                 },
                 // 1 day in milliseconds = 24 * 60 * 60 * 1000 = 86,400,000
                 'minRangeSize': 86400000
               }
             },
             // Initial range: 2012-02-09 to 2012-03-20.
              'state': {'range': {'start': new Date(data_inicial_cortada[2],(data_inicial_cortada[1]-1),data_inicial_cortada[0]), 
                                 'end': new Date(data_final_cortada[2],(data_final_cortada[1]-1),data_final_cortada[0])}}
           });
        
           var chart = new google.visualization.ChartWrapper({
             'chartType': 'LineChart',
             'containerId': 'chart-pequeno',
             'options': {
               // Use the same chart area width as the control for axis alignment.
               'chartArea': {'height': '136px', 'width': '350px'},
               'hAxis': {'slantedText': false},
               'legend': {'position': 'none'},
               //'vAxis': { 'format': '\'R$\' #' },
               'pointSize':5
             },
             // Convert the first column from 'date' to 'string'.
             'view': {
               'columns': [
                 {
                   'calc': function(dataTable, rowIndex) {
                     return dataTable.getFormattedValue(rowIndex, 0);
                   },
                   'type': 'string'
                 }, 1]
             }
           });

           var data = new google.visualization.DataTable();
           data.addColumn('date', 'Date');
           data.addColumn('number', 'Fechamento');

            $.each(historicos, function(i){
                
                data_cortada = historicos[i].data.split("/");
                date = new Date(data_cortada[2],(data_cortada[1]-1),data_cortada[0]);
                data.addRow([date, historicos[i].fechamento]);
            });

            var formatter = new google.visualization.DateFormat({pattern: "dd MMM yy"});
            formatter.format(data, 0);

            var formatter = new google.visualization.NumberFormat({
                fractionDigits: 2
                //prefix: 'R$ '
            });
            formatter.format(data, 1);

            dashboard.bind(control, chart);
            dashboard.draw(data);
        }
    });

    function string_to_slug(str) {
        str = str.replace(/^\s+|\s+$/g, ''); // trim

        // remove accents, swap ñ for n, etc
        var from = "ÃÁÉÍÓÚÇãàáäâèéëêìíïîòóöôõùúüûñç·/_,:;";
        var to   = "AAEIOUCaaaaaeeeeeiiiiooooouuuunc------";

        for (var i=0, l=from.length ; i<l ; i++) {
            str = str.replace(new RegExp(from.charAt(i), 'g'), to.charAt(i));
        }

        str = str.replace(/[^A-Za-z0-9 -]/g, '') // remove invalid chars
        str = str.replace(/\s+/g, '-') // collapse whitespace and replace by -
        str = str.replace(/-+/g, '-'); // collapse dashes

        return str.toLowerCase();
    }

    function truncar(texto,limite){
        if(texto.length>limite){
            limite--;
            last = texto.substr(limite-1,1);
            while(last!=' ' && limite > 0){
                limite--;
                last = texto.substr(limite-1,1);
            }
            last = texto.substr(limite-2,1);
            if(last == ',' || last == ';'  || last == ':'){
                 texto = texto.substr(0,limite-2) + '...';
            } else if(last == '.' || last == '?' || last == '!'){
                 texto = texto.substr(0,limite-1);
            } else {
                 texto = texto.substr(0,limite-1) + '...';
            }
        }
        return texto;
    }
