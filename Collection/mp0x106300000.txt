		function fcnConsultaPrecoByQTD(numQtd, numQtdCopia, bitExibeMSG){
			var txtAdicionais = "";

			if($("DIV#innerQuantidades").length>0) $("DIV#innerQuantidades").addClass("loading");

			$("DIV#innerOpcoes input:checkbox.adicionais").each(function () {
				if (this.checked == true) txtAdicionais += ((txtAdicionais != "") ? (",") : ("")) + $(this).attr("value");
			});

			var txtURL = "/loja/nav/produtos_detalhes_preco.asp?idProduto=8NQFNCPJ&v=3";
			txtURL += "&numQtd=" + numQtd;
			txtURL += "&numCopias=" + numQtdCopia;
			if($("FORM#frmDetalhesProduto input:text#txt_m2_altura_cm").length==1) txtURL += "&numM2Altura=" + $("FORM#frmDetalhesProduto input:text#txt_m2_altura_cm").val();
			if($("FORM#frmDetalhesProduto input:text#txt_m2_largura_cm").length==1) txtURL += "&numM2Largura=" + $("FORM#frmDetalhesProduto input:text#txt_m2_largura_cm").val();
			txtURL += "&txtAdicionais=" + escape(txtAdicionais);
			txtURL += "&noCache=" + escape(Date.now());

			$.ajax({
				type: "GET",
				url: txtURL,
				beforeSend: function () {
					if($("DIV#innerQuantidades DIV#innerListaQTDs P#innerErro").length>0) $("DIV#innerQuantidades DIV#innerListaQTDs P#innerErro").remove();
					if($("DIV#innerQuantidades").length>0) $("DIV#innerQuantidades DIV#innerPreco_" + numQtd).addClass('dp-fluid').addClass('loading');
					if($("P#innerM2Erro").length>0) $("P#innerM2Erro").html("");
				},
				success: function (objJson) {
					objJson = $.parseJSON(objJson)
					//console.log(objJson);

					if(objJson.bitErro=="0" && isNaN(objJson.numValorTotal)){
						objJson.bitErro="99";
						objJson.txtErro="Não foi possível encontrar valores para esta quantidade informada.";
					}

					if(objJson.bitErro!="0"){
						if(objJson.txtErro!=""){
							if(bitExibeMSG=="1") $.colorbox({ html: '<div class="message error">' + objJson.txtErro + '</div>', fixed: true, opacity: 0.3, width: 300, height: 200 });
							if(objJson.bitErro=="10") if($("P#innerM2Erro").length>0) $("P#innerM2Erro").html(objJson.txtErro);
						}else{
							if(bitExibeMSG=="1") $.colorbox({ html: '<div class="message error">A quantidade informada não é válida.</div>', fixed: true, opacity: 0.3, width: 300, height: 200 });
						}
						if($("DIV#innerQuantidades DIV#innerListaQTDs").length>0) $("DIV#innerQuantidades DIV#innerListaQTDs DIV#innerPreco_" + objJson.numQtd).remove();
						if($("DIV#innerQuantidades DIV#innerListaQTDs DIV").length<=0) $("DIV#innerQuantidades DIV#innerListaQTDs").append("<P id='innerErro' style='color:#ff0000;'>" + objJson.txtErro + "</P>");
					}else{
						if($("DIV#innerQuantidades").length>0){
							if(!isNaN(objJson.numValorTotal)){
								$("DIV#innerQuantidades DIV#innerListaQTDs DIV#innerPreco_" + objJson.numQtd).attr("numtotal", objJson.numValorTotal);
								$("DIV#innerQuantidades DIV#innerPreco_" + objJson.numQtd).find('.price').text("R$ " + formatNumber(objJson.numValorTotal, 2, ",", "."));
							}
							if(!isNaN(objJson.numValorUnit)) $("DIV#innerQuantidades DIV#innerPreco_" + objJson.numQtd).find('.price-unity').text("R$ " + formatNumber(objJson.numValorUnit, 2, ",", ".") + '/' + objJson.txtUnidadeMedida);
							$("DIV#innerQuantidades DIV#innerPreco_" + objJson.numQtd).removeClass('dp-fluid').removeClass('loading');
						}
						if($("DIV#innerQuantidades DIV.novaCaixa").length>0){ // Processa somente se for "outra quantidade"
							$("DIV#innerQuantidades INPUT.itensQTD").each(function(){
								$(this).checked = false;
								$(this).prop("checked", false);
							});
							$("DIV#innerQuantidades INPUT#txt_qtd_" + objJson.numQtd).checked = true;
							$("DIV#innerQuantidades INPUT#txt_qtd_" + objJson.numQtd).prop("checked", true);
						}
						
						if($("DIV#innerMedida INPUT#numtotal").length>0){
							$("DIV#innerMedida INPUT#numtotal").val(objJson.numValorTotal);
						}
					}
					if(!isNaN(objJson.numM2)){
						if($("DIV#innerMedida").length>0) $("DIV#innerMedida SPAN#innerTotalM2").html(formatNumber(objJson.numM2, 2, ",", "."));
					}
				},
				complete: function (objHTTP, txtStatus) {
					numTimer = null;
					fcnSimulaFrete();
					fcnCalculaResumo();
					fcnCalculaCopias();
					if($("DIV#innerQuantidades").length>0) $("DIV#innerQuantidades").removeClass("loading");
				}
			});
		}

		function fcnCalculaCopias() {
			var numTotalVL = 0, numTotalCalculo = 0;

				});

			$("DIV#innerQTDCopias DIV.list").removeClass('dp-fluid').removeClass('loading');
