            if (email != "" || nome != "") {
                if (filtro.test(email)) {

                    $.post(url, self.serialize(), function(data) {

                        if ( data === 'sucesso' ) {
                            $(".newsletter-feedback-modal").children('p:last').hide().parent().slideDown('fast');
                            self[0].reset();
                        } else {
                            var dataJson = JSON.parse(data),
                                dataCodigo = dataJson.codigo,
                                dataMSG = dataJson.msg;

                            if ( dataCodigo === 214 ) {
                                $(".newsletter-feedback-modal").children('p:first').hide().parent().slideDown('fast');
                            } else {
                                alert( 'Erro '+dataCodigo+' ao cadastrar o e-mail' );
                            }
                        }
                        self.find('.btn-submit').removeClass('off').prop("disabled", false);
                    });

                } else {
                    $("#email-modal").val("Este endereço de email não é válido!");
                    self.find('.btn-submit').removeClass('off').prop("disabled", false);
                    return false;
                } // if regex email
            } else { // if email || nome
                $("#email-modal").val('Digite um email!');
                self.find('.btn-submit').removeClass('off').prop("disabled", false);
                return false;
            }
        });

    })
